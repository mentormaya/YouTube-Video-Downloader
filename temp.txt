import threading

def target_function(num):
    print(f"{num}th target job done...")

threads = []

for i in range(50):
    t = threading.Thread(target = target_function, args = (i,))
    t.daemon = True
    threads.append(t)

for i in range(50):
    threads[i].start()

for i in range(50):
    threads[i].join()
